43 static glm::vec4 CorrectGamma(const glm::vec4 &inputColor, float gamma)
44 {
45  glm::vec4 outputColor;
46  outputColor[0] = powf(inputColor[0], 1.0f / gamma);
47  outputColor[1] = powf(inputColor[1], 1.0f / gamma);
48  outputColor[2] = powf(inputColor[2], 1.0f / gamma);
49  outputColor[3] = inputColor[3];
50 
51  return outputColor;
52 }
